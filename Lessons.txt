function updateProgress() {
  const pct = (done.filter(Boolean).length / steps.length) * 100;
  document.getElementById('progress-bar').style.width = pct + '%';
  document.querySelectorAll('.tab').forEach((t, i) => {
    t.classList.toggle('done', done[i]);
  });
}

function showStep(index) {
  currentStep = index;
  const s = steps[index];
  document.getElementById('step-number').innerText = `Step ${index + 1} of ${steps.length}`;
  document.getElementById('section-label').innerText = s.section;
  document.getElementById('step-instructions').innerHTML = s.instructions;
  document.querySelectorAll('.tab').forEach((t, i) => t.classList.toggle('active', i === index));
  document.getElementById('check-result').innerText = '';
  // ← textarea is NEVER touched — code stays as the student builds it
}

function checkStep() {
  const code = document.getElementById('step-code').value;
  const ok = steps[currentStep].check(code);
  if (ok) {
    done[currentStep] = true;
    updateProgress();
    document.getElementById('check-result').innerText = '✅ Correct! Move to the next step.';
  } else {
    document.getElementById('check-result').innerText = '❌ Not quite — re-read the instructions and check your code.';
  }
}

function nextStep() {
  if (currentStep < steps.length - 1) showStep(currentStep + 1);
